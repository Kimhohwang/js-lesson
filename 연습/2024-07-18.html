/* 종류-01. 사칙 연산 */

console.log(1 + 2);

console.log(1 - 2);

console.log(1 * 2);

console.log(10 / 2); // => 5

console.log(11 % 3); // => 2

console.log(2 ** 3); // => 8

console.log(1 + 2 * 2); // => 5

연산 기호 종류 +, -, *, /, %, **

더하기, 빼기, 곱하기, 나누기, 나눈후 나머지값, 몇승

/* 종류-02. 문자열 연산 */

console.log("my" + "message"); // => "mymessage"

console.log(2 + 2 + "1"); // => "41"

console.log("2" + "3"); // => "23"

문자2와 문자3을 더해서 문자 23이라는 결과 출력

미션 내용

두 개의 숫자를 입력하여 연산하는 계산기를 구현한다.

연산의 종류는 +, *, % 3개의 종류가 있다.

연산의 종류를 판별하여 적절한 연산을 수행 후 결과를 출력한다.

만약 하나라도 숫자 이외의 입력이 발생하면 연산의 결과는 null을 출력한다.

만약 하나라도 지정 연산자 3개 이외의 입력이 발생하면 연산의 결과는 null을 출력한다.

  두개의 숫자 정의
  연산의 종류 정의

  하나라도 숫자가 아니라면 null값 출력
  하나라도 지정된 연산자가 아니라면 null값 출력

  연산 종류 판별하여 적절한 연산을 수행한다.

let a1 = 10;

let a2 = 30;

const operator = "+";

const isAisNumber = typeof a1 == "number" && typeof a2 == "number";

const isOperatorValid = operator == "+" || operator == "*" || operator == "%";

if (!isAisNumber && !isOperatorValid) {

console.log(null);

} else {

if (operator == "+") {

console.log(result = a1 + a2);

} else if (operator == "%") {

console.log(result = a1 % a2);

} else if (operator == "*") {

console.log(result = a1 * a2);

}

}

console.log(result); 

    변수명 a1에 숫자 10의 값을 부여한다;
    변수명 a2에 숫자 30의 값을 부여한다;
    상수명 operator에 문자 + 값을 부여한다;
    상수명 isAisNumber는 변수 a1, a2의 타입이 number여야한다.
    상수명 isOperatorValid는 상수명 operator의 값이 +, *, % 3가지 값중 하나라도 포함 되어야한다.

    만약 isAisNumber의 값이 해당되지 않고 그리고 isOperatorValid의 값이 해당되지 않는다면
    null값을 출력한다
    null값이 아니면
    만약 상수명 operator가 문자 + 와 같을때는
    result라는 단어의 값은 a1 + a2 이다

    result라는 값은 a1 * a2 이다
    또 아니고 만약 상수명 operator가 문자 % 와 같을때는
